--- Optional: download yolo26n pretrained if missing
python - <<'PY'
from pathlib import Path
import urllib.request
p = Path('d:/git/projects/teknim_kontak_yay3/yolo26n.pt')
if not p.exists():
    print('Downloading yolo26n.pt...')
    url = 'https://github.com/ultralytics/assets/releases/download/v0.0.0/yolo26n.pt'
    urllib.request.urlretrieve(url, str(p))
    print('Downloaded to', p)
else:
    print('yolo26n.pt already exists')
PY

--- Training (direct)
# Simple direct training with Ultralytics (recommended for one-shot training):
yolo detect train model=yolo26n.pt data=projects/teknim_kontak_yay3/data.yaml epochs=100 imgsz=640
